% MACROS %
\newcommand{\uvec}[1]{\boldsymbol{\hat{\textbf{#1}}}}
\newcommand{\evalmath}[1]{\pgfmathparse{#1}\pgfmathprintnumber{\pgfmathresult}}

\newcommand{\transform}[3]{
  T(#1) &=
  \begin{bmatrix}
    0 & -1 \\
    -1 & 0
  \end{bmatrix}
  \begin{bmatrix}
    #2 \\ #3
  \end{bmatrix} =
  \begin{bmatrix}
    (0 \cdot #2) + (-1 \cdot #3) \\
    (-1 \cdot #2) + (0 \cdot #3)
  \end{bmatrix} =
  \begin{bmatrix}
    \evalmath{0 * #2} + \evalmath{-1 * #3} \\
    \evalmath{-1 * #2} + \evalmath{0 * #3}
  \end{bmatrix} =
  \begin{bmatrix}
    \evalmath{0 * #2 + -1 * #3} \\
    \evalmath{-1 * #2 + 0 * #3} \\
  \end{bmatrix} = #1' \\
}
